import React, { Component } from 'react';
import { Button, Divider, Header } from 'semantic-ui-react'
import socket from "../../socket.js";


/**
 * @class WorkerToolBox
 * @extends {React.Component}
 * @description Class for displaying the Toolbox for initiation the dynamic topic detection
 */
class WorkerToolBox extends Component {

    state = {
        topicWorker: null,
        sentimentWorker: null,
    };


    /**
     * @function sendMessage
     * @param  {Object} message Message to send to the server
     * @description Sends a message to the server, either start or stop the worker classes
     * @memberof WorkerToolBox
     * @return {void}
     */
    sendMessage = message => {
        socket.emit('client:get', message)
    }

    render() {
        const { topicWorker, sentimentWorker } = this.state;
        return (
            <div>
                <Button.Group>
                    {topicWorker ? <Button negative>Stop topic worker</Button> : <Button positive>Start topic worker</Button>}
                    {sentimentWorker ? <Button negative>Stop sentiment worker</Button> : <Button positive>Start sentiment worker</Button>}
                </Button.Group>
            </div>


        );
    }
}

export default WorkerToolBox;
